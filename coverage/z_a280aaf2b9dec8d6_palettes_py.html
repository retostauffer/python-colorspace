1249 def _chroma_trajectory(self, i, p1, c1, c2, cmax): 

1250 """Helper function: Calculate linear or triangle trajectory for chroma dimension. 

1251 

1252 Args: 

1253 i (numpy array; float): Position across the palette, a sequence 

1254 of values between 1 and 0. For diverging palettes this function 

1255 is called twice, once for 1 to 0.5, and once for <0.5 t0 0. 

1256 p1 (float): Power parameter p1. 

1257 c1 (float): Chroma value of the left end of the color palette. 

1258 c2 (float): Chroma value of the right end of the color palette. 

1259 cmax (float or None): Max choma value. 

1260 

1261 Returns: 

1262 numpy array: Linear trajectory for the chroma color dimension. 

1263 """ 

1264 

1265 def _linear_trajectory(i, c1, c2): 

1266 return c2 - (c2 - c1) * i 

1267 

1268 def _triangle_trajectory(i, j, c1, c2, cmax): 

1269 from numpy import where, abs, linspace 

1270 res = where(i <= j, 

1271 c2 - (c2 - cmax) * i / j, 

1272 cmax - (cmax - c1) * abs((i - j) / (1 - j))) 

1273 #print(res) 

1274 return res 

1275 

1276 if cmax is None: 

1277 C = _linear_trajectory(i**p1, c1, c2) 

1278 else: 

1279 # Calculate the position of the triangle point 

1280 j = 1. / (1. + abs(cmax - c1) / abs(cmax - c2)) 

1281 if not j is None and (j <= 0. or j >= 1.): j = None 

1282 

1283 if j is None: C = _linear_trajectory(i**p1, c1, c2) 

1284 else: C = _triangle_trajectory(i**p1, j, c1, c2, cmax)

1292class qualitative_hcl(hclpalette): 

1293 """Qualitative HCL Color Palettes 

1294 

1295 The HCL (hue-chroma-luminance) color model is a perceptual color model 

1296 obtained by using polar coordinates in CIELUV space 

1297 (i.e., :py:class:`polarLUV <colorspace.colorlib.polarLUV>`), 

1298 where steps of equal size correspond to approximately equal perceptual 

1299 changes in color. By taking polar coordinates the resulting three 

1300 dimensions capture the three perceptual axes very well: hue is the type of 

1301 color, chroma the colorfulness compared to the corresponding gray, and 

1302 luminance the brightness. This makes it relatively easy to create balanced 

1303 palettes through trajectories in this HCL space. In contrast, in the more 

1304 commonly-used ‘HSV’ (hue-saturation-value) model (a simple transformation 

1305 of ‘RGB’), the three axes are confounded so that luminance changes along 

1306 with the hue leading to very unbalanced palettes. 

1307 

1308 `qualitative_hcl` distinguishes the underlying categories by a 

1309 sequence of hues while keeping both chroma and luminance constant to give 

1310 each color in the resulting palette the same perceptual weight. Thus, `h` 

1311 should be a pair of hues (or equivalently `h1` and `h2` can be used) with 

1312 the starting and ending hue of the palette. Then, an equidistant sequence

2811 argument `n` (number of colors), see examples. 

2812

2818 Args: 

2819 s (float, int): saturation value, a value in `[0., 1.]`. Defaults to `1.0`. 

2820 v (float, int): value, a value in `[0., 1.]`. Defaults to `1.0`.

2822 the rainbow begins. Defaults to `0.`. Can be a function with one input

2825 the rainbow ends. Defaults to `0.`. Can be a function with one input

2828 *args: Unused.

2845 >>> rainbow()(10) 

2846 >>>

2848 >>> p = rainbow(start = lambda n: 1 / n, end = lambda n: 1 - 1 / n)
